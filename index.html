name: Sandbox CI/CD on: push: branches: - main jobs: build-and-deploy: runs-on: [self-hosted, Linux, X64, DockerServer] steps: - name: Checkout code uses: actions/checkout@v4 - name: Show info run: | echo "Repo: ${{ github.repository }}" echo "Branch: ${{ github.ref }}" echo "Commit: ${{ github.sha }}" - name: Build artifacts run: | echo "Building project..." rm -rf build mkdir -p build cp *.html build/ 2>/dev/null || true - name: Deploy to server folder run: | echo "Deploying to /home/ubuntu/bd-testing" DEPLOY_DIR="/home/ubuntu/bd-testing" rm -rf $DEPLOY_DIR/* mkdir -p $DEPLOY_DIR cp -r build/* $DEPLOY_DIR/ echo "Deployment completed successfully" echo "Files deployed to $DEPLOY_DIR"
